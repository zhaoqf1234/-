// 使用 MERGE 检查并创建 "墨家禁地" 节点，如果不存在则创建
MERGE (mjjd:Location {name: '墨家禁地'})

// 使用 MERGE 检查并创建 "墨家村" 节点，如果不存在则创建
MERGE (mjc:Location {name: '墨家村'})

// 连接墨家禁地到墨家村
CREATE (mjjd)-[:连接]->(mjc)

// 创建 NPC 节点并与墨家禁地连接
CREATE (npc1:NPC {name: '机关人', location: '15,99'})
CREATE (npc2:NPC {name: '巡逻的机关人', location: '随机移动'})
CREATE (npc3:NPC {name: '超级巫医', location: '131,69'})

// 建立墨家禁地与所有NPC的"包含"关系
CREATE (mjjd)-[:包含]->(npc1)
CREATE (mjjd)-[:包含]->(npc2)
CREATE (mjjd)-[:包含]->(npc3)
