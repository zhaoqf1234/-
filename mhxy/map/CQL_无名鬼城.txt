// 使用 MERGE 检查并创建 "无名鬼城" 节点，如果不存在则创建
MERGE (wmgc:Location {name: '无名鬼城'})

// 使用 MERGE 检查并创建 "地狱迷宫" 节点，如果不存在则创建
MERGE (dymg:Location {name: '地狱迷宫'})

// 连接无名鬼城到地狱迷宫
CREATE (wmgc)-[:连接]->(dymg)

// 创建 NPC 节点并与无名鬼城连接
CREATE (npc1:NPC {name: '鬼将军', location: '31,19'})
CREATE (npc2:NPC {name: '超级巫医', location: '61,102'})

// 建立无名鬼城与所有NPC的"包含"关系
CREATE (wmgc)-[:包含]->(npc1)
CREATE (wmgc)-[:包含]->(npc2)
